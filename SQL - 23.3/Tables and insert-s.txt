CREATE TABLE Newspaper (
	IdOfNewspaper INT NOT NULL,
	NewspaperName VARCHAR(20) NOT NULL,
	PublishDate DATETIME,
	ArticleName VARCHAR(20), 
	CompanyName VARCHAR(20),
	PRIMARY KEY (IdOfNewspaper),
);

CREATE TABLE Article (
	IdOfArticle INT NOT NULL,
	ArticleName VARCHAR(20) NOT NULL,
	AuthorId VARCHAR(15),
	IdOfNewspaper INT,
	PRIMARY KEY (IdOfArticle),
	FOREIGN KEY (IdOfNewspaper) REFERENCES Newspaper(IdOfNewspaper),
);

CREATE TABLE Company (
	CompanyId INT NOT NULL,
	CompanyName VARCHAR(20) NOT NULL,
	NewspaperId INT,
	PRIMARY KEY (CompanyId),
	FOREIGN KEY (NewspaperId) REFERENCES Newspaper (IdOfNewspaper)
);

CREATE TABLE Journalist (
	JournalistId INT NOT NULL,
	LastName VARCHAR(20) NOT NULL,
	ArticleId INT,
	PRIMARY KEY (JournalistId),
	FOREIGN KEY (ArticleId) REFERENCES Article (IdOfArticle)
);
